25 export async function PATCH(req: NextRequest) {
26   const user = await getSessionUser();
27 
28   if (!user) {
29     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
30   }
31 
32   const data = await req.json();
33 
34   const name = String(data.name ?? "").trim();
35   const email = String(data.email ?? "").toLowerCase().trim();
36   const cpf = normalizeCpf(String(data.cpf ?? "").trim());
37   const currentPassword = String(data.currentPassword ?? "");
38   const newPassword = String(data.newPassword ?? "");
39   const confirmNewPassword = String(data.confirmNewPassword ?? "");
40 
41   if (!name || !email || !cpf) {
42     return NextResponse.json(
43       { error: "Nome, e-mail e CPF são obrigatórios." },
44       { status: 400 }
45     );
46   }
47 
48   if (cpf.length !== 11) {
49     return NextResponse.json(
50       { error: "CPF deve conter 11 dígitos." },
51       { status: 400 }
52     );
53   }
54 
55   const existingByEmail = await prisma.user.findFirst({
56     where: {
57       email,
58       NOT: { id: user.id },
59     },
60   });
61 
62   if (existingByEmail) {
63     return NextResponse.json(
64       { error: "Este e-mail já está em uso." },
65       { status: 409 }
66     );
67   }
68 
69   const existingByCpf = await prisma.user.findFirst({
70     where: {
71       cpf,
72       NOT: { id: user.id },
73     },
74   });
75 
76   if (existingByCpf) {
77     return NextResponse.json(
78       { error: "Este CPF já está em uso." },
79       { status: 409 }
80     );
81   }
82 
83   const wantsPasswordChange =
84     currentPassword.length > 0 ||
85     newPassword.length > 0 ||
86     confirmNewPassword.length > 0;
87 
88   let passwordHash: string | undefined;
89 
90   if (wantsPasswordChange) {
91     if (!currentPassword || !newPassword || !confirmNewPassword) {
92       return NextResponse.json(
93         {
94           error:
95             "Para trocar a senha, preencha senha antiga, senha nova e repetir senha nova.",
96         },
97         { status: 400 }
98       );
99     }
100 
101     const dbUser = await prisma.user.findUnique({
102       where: { id: user.id },
103     });
104 
105     if (!dbUser) {
106       return NextResponse.json({ error: "Usuário não encontrado." }, { status: 404 });
107     }
108 
109     const oldPasswordMatches = await verifyPassword(
110       currentPassword,
111       dbUser.passwordHash
112     );
113 
114     if (!oldPasswordMatches) {
115       return NextResponse.json(
116         { error: "A senha antiga está incorreta." },
117         { status: 400 }
118       );
119     }
120 
121     if (newPassword !== confirmNewPassword) {
122       return NextResponse.json(
123         { error: "A nova senha e a confirmação não conferem." },
124         { status: 400 }
125       );
126     }
127 
128     if (newPassword.length < 6) {
129       return NextResponse.json(
130         { error: "A nova senha deve ter pelo menos 6 caracteres." },
131         { status: 400 }
132       );
133     }
134 
135     passwordHash = await hashPassword(newPassword);
136   }
137 
138   const updated = await prisma.user.update({
139     where: { id: user.id },
140     data: {
141       name,
142       email,
143       cpf,
144       ...(passwordHash ? { passwordHash } : {}),
145     },
146   });
147 
148   return NextResponse.json({
149     user: {
150       id: updated.id,
151       name: updated.name,
152       email: updated.email,
153       cpf: updated.cpf ?? "",
154       isTeacher: updated.isTeacher,
155     },
156     passwordChanged: Boolean(passwordHash),
157   });
158 }
