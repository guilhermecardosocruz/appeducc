5 async function ensureAttendanceAccess(userId: string, attendanceId: string) {
6   const attendance = await prisma.attendance.findUnique({
7     where: { id: attendanceId },
8     include: {
9       class: {
10         include: {
11           school: true,
12         },
13       },
14       presences: {
15         include: {
16           student: {
17             select: {
18               id: true,
19               name: true,
20             },
21           },
22         },
23         orderBy: {
24           student: {
25             name: "asc",
26           },
27         },
28       },
29     },
30   });
31 
32   if (!attendance) return null;
33 
34   const membership = await prisma.schoolMember.findUnique({
35     where: {
36       userId_schoolId: {
37         userId,
38         schoolId: attendance.class.schoolId,
39       },
40     },
41   });
42 
43   if (!membership) return null;
44 
45   return attendance;
46 }

67 export async function PATCH(
68   req: NextRequest,
69   { params }: { params: Promise<{ attendanceId: string }> }
70 ) {
71   const user = await getSessionUser();
72   if (!user) {
73     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
74   }
75

81   }
82 
83   const data = await req.json();

86   const presences = Array.isArray(data.presences) ? data.presences : [];

1 import { NextRequest, NextResponse } from "next/server";

5 async function ensureClassAccess(userId: string, classId: string) {
6   const foundClass = await prisma.class.findUnique({
7     where: { id: classId },
8     include: {
9       school: true,
10       students: {
11         orderBy: { name: "asc" },
12       },
13     },
14   });
15 
16   if (!foundClass) return null;
17 
18   const membership = await prisma.schoolMember.findUnique({
19     where: {
20       userId_schoolId: {
21         userId,
22         schoolId: foundClass.schoolId,
23       },
24     },
25   });
26 
27   if (!membership) return null;
28 
29   return foundClass;
30 }

32 export async function GET(
33   _req: NextRequest,
34   { params }: { params: Promise<{ classId: string }> }
35 ) {
36   const user = await getSessionUser();
37   if (!user) {
38     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
39   }
40 
41   const { classId } = await params;
42   const foundClass = await ensureClassAccess(user.id, classId);
43 
44   if (!foundClass) {
45     return NextResponse.json({ error: "Forbidden" }, { status: 403 });
46   }
47 
48   const attendances = await prisma.attendance.findMany({
49     where: { classId },
50     include: {
51       presences: {
52         select: {
53           id: true,
54           present: true,
55         },
56       },
57     },
58     orderBy: [{ lessonDate: "desc" }, { createdAt: "desc" }],
59   });
60 
61   return NextResponse.json(attendances);
62 }

64 export async function POST(
65   req: NextRequest,
66   { params }: { params: Promise<{ classId: string }> }
67 ) {
68   const user = await getSessionUser();
69   if (!user) {
70     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
71   }
72 
73   const { classId } = await params;
74   const foundClass = await ensureClassAccess(user.id, classId);
75 
76   if (!foundClass) {
77     return NextResponse.json({ error: "Forbidden" }, { status: 403 });
78   }
79 
80   const data = await req.json();
81   const title = String(data.title ?? "").trim();
82   const lessonDateRaw = String(data.lessonDate ?? "").trim();
83   const presencesInput = Array.isArray(data.presences) ? data.presences : [];
84 
85   if (!title) {
86     return NextResponse.json(
87       { error: "Missing attendance title" },
88       { status: 400 }
89     );
90   }
91 
92   if (!lessonDateRaw) {
93     return NextResponse.json(
94       { error: "Missing attendance date" },
95       { status: 400 }
96     );
97   }
98 
99   const lessonDate = new Date(`${lessonDateRaw}T12:00:00`);
100 
101   if (Number.isNaN(lessonDate.getTime())) {
102     return NextResponse.json(
103       { error: "Invalid attendance date" },
104       { status: 400 }
105     );
106   }
107 
108   const presentByStudentId = new Map<string, boolean>();
109   for (const item of presencesInput) {
110     const studentId = String(item?.studentId ?? "");
111     if (!studentId) continue;
112     presentByStudentId.set(studentId, Boolean(item?.present));
113   }
114 
115   const attendance = await prisma.attendance.create({
116     data: {
117       title,
118       classId,
119       lessonDate,
120       presences: {
121         create: foundClass.students.map((student) => ({
122           studentId: student.id,
123           present: presentByStudentId.get(student.id) ?? false,
124         })),
125       },
126     },
127     include: {
128       presences: {
129         include: {
130           student: {
131             select: {
132               id: true,
133               name: true,
134             },
135           },
136         },
137         orderBy: {
138           student: {
139             name: "asc",
140           },
141         },
142       },
143     },
144   });
145 
146   return NextResponse.json(attendance, { status: 201 });
147 }
