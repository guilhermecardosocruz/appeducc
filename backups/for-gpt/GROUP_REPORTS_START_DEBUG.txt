63       }
64     } finally {
65       setLoadingList(false);

69   async function refreshMembers() {

72     const res = await fetch(`/api/groups/${groupId}/members`, {
73       method: "GET",
74       headers: { "Content-Type": "application/json" },

84   async function handleRemoveMember(

1 import { notFound } from "next/navigation";
2 import GroupDetailClient from "@/components/GroupDetailClient";
3 import { getSessionUser } from "@/lib/auth";

6 type PageProps = {
7   params: Promise<{ groupId: string }>;
8 };
9 
10 export default async function GroupPage({ params }: PageProps) {
11   const user = await getSessionUser();
12 
13   if (!user) {
14     notFound();
15   }
16 
17   const { groupId } = await params;
18 
19   const [groupMembership, schoolMembershipsInGroup, group] = await Promise.all([
20     prisma.groupMember.findUnique({
21       where: {
22         userId_groupId: {
23           userId: user.id,
24           groupId,
25         },
26       },
27     }),
28     prisma.schoolMember.findMany({
29       where: {
30         userId: user.id,
31         school: {
32           groupId,
33         },
34       },
35       select: {
36         schoolId: true,
37         role: true,
38       },
39     }),
40     prisma.group.findUnique({
41       where: { id: groupId },
42       include: {
43         schools: {
44           include: {
45             _count: {
46               select: { classes: true },
47             },
48           },
49           orderBy: { name: "asc" },
50         },
51         members: {
52           include: {
53             user: {
54               select: {
55                 id: true,
56                 name: true,
57                 email: true,
58                 cpf: true,
59                 isTeacher: true,
60                 createdAt: true,
61               },
62             },
63           },
64           orderBy: [{ role: "asc" }, { createdAt: "asc" }],
65         },
66       },
67     }),
68   ]);
69 
70   if (!group) {
71     notFound();
72   }
73 
74   const hasAccess =
75     Boolean(groupMembership) || schoolMembershipsInGroup.length > 0;
76 
77   if (!hasAccess) {
78     notFound();
79   }
80 
81   const canManageMembers =
82     groupMembership?.role === "OWNER" || groupMembership?.role === "MANAGER";
83 
84   const allowedSchoolIds = groupMembership
85     ? null
86     : new Set(schoolMembershipsInGroup.map((item) => item.schoolId));
87 
88   const schools = group.schools
89     .filter((school) => {
90       if (!allowedSchoolIds) return true;
91       return allowedSchoolIds.has(school.id);
92     })
93     .map((school) => ({
94       id: school.id,
95       name: school.name,
96       createdAt: school.createdAt.toISOString(),
97       _count: {
98         classes: school._count.classes,
99       },
100     }));
101 
102   const members = group.members.map((item) => ({
103     userId: item.user.id,
104     name: item.user.name,
105     email: item.user.email,
106     cpf: item.user.cpf,
107     isTeacher: item.user.isTeacher,
108     role: item.role,
109     memberSince: item.createdAt.toISOString(),
110     createdAt: item.user.createdAt.toISOString(),
111   }));
112 
113   return (
114     <GroupDetailClient
115       groupId={group.id}
116       groupName={group.name}
117       initialSchools={schools}
118       initialMembers={members}
119       canManageMembers={canManageMembers}
120       currentUserId={user.id}
121     />
122   );
123 }
