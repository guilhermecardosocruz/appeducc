16 export async function createSession(userId: string, baseUrl: string) {
17   const expiresAt = new Date();
18   expiresAt.setDate(expiresAt.getDate() + SESSION_DAYS);
19 
20   const session = await prisma.session.create({
21     data: {
22       userId,
23       expiresAt,
24     },
25   });
26 
27   const cookieStore = await cookies();
28 
29   cookieStore.set(SESSION_COOKIE_NAME, session.id, {
30     httpOnly: true,
31     secure: true,
32     sameSite: "lax",
33     path: "/",
34     expires: expiresAt,
35   });
36 
37   return new URL("/dashboard", baseUrl);
38 }

40 export async function destroySession() {
41   const cookieStore = await cookies();
42   const sessionId = cookieStore.get(SESSION_COOKIE_NAME)?.value;
43 
44   if (sessionId) {
45     await prisma.session.deleteMany({
46       where: { id: sessionId },
47     });
48 
49     cookieStore.set(SESSION_COOKIE_NAME, "", {
50       httpOnly: true,
51       secure: true,
52       sameSite: "lax",
53       path: "/",
54       maxAge: 0,
55     });
56   }
57 }
58 
59 export async function getSessionUser() {
60   const cookieStore = await cookies();
61   const sessionId = cookieStore.get(SESSION_COOKIE_NAME)?.value;
62 
63   if (!sessionId) return null;
64 
65   const session = await prisma.session.findUnique({
66     where: { id: sessionId },
67     include: { user: true },
68   });
69 
70   if (!session) return null;
71   if (session.expiresAt < new Date()) {
72     return null;
73   }
74 
75   return session.user;
76 }
