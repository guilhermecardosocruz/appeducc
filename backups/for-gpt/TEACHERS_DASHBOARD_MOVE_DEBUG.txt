3 import Link from "next/link";
4 import { useState } from "react";
5 import CreateGroupModal from "./CreateGroupModal";
6 import CreateTeacherModal from "./CreateTeacherModal";

8 type Group = {
9   id: string;
10   name: string;
11   createdAt: string;
12   _count: {
13     schools: number;
14   };
15 };
16 
17 type Teacher = {
18   id: string;
19   name: string;
20   email: string;
21   isTeacher: boolean;
22   createdAt: string;
23   _count: {
24     classes: number;
25   };
26 };
27 
28 type Props = {
29   initialGroups: Group[];
30   initialTeachers: Teacher[];
31   userName?: string | null;
32   userEmail?: string | null;

34 
35 export default function GroupsDashboardClient({
36   initialGroups,
37   initialTeachers,
38   userName,
39   userEmail,
40 }: Props) {
41   const [groups, setGroups] = useState<Group[]>(initialGroups);
42   const [teachers, setTeachers] = useState<Teacher[]>(initialTeachers);
43   const [openGroupModal, setOpenGroupModal] = useState(false);
44   const [openTeacherModal, setOpenTeacherModal] = useState(false);
45   const [loadingGroupList, setLoadingGroupList] = useState(false);
46   const [loadingTeacherList, setLoadingTeacherList] = useState(false);
47 
48   async function refreshGroups() {
49     setLoadingGroupList(true);
50     try {
51       const res = await fetch("/api/groups", {
52         method: "GET",
53         headers: { "Content-Type": "application/json" },
54         cache: "no-store",
55       });
56 
57       if (res.ok) {
58         const data = (await res.json()) as Group[];
59         setGroups(
60           data.map((g) => ({
61             id: g.id,
62             name: g.name,
63             createdAt: g.createdAt,
64             _count: { schools: g._count?.schools ?? 0 },
65           }))
66         );
67       } else {
68         console.error("Erro ao carregar grupos");
69       }
70     } catch (error) {
71       console.error("Erro ao buscar grupos", error);
72     } finally {
73       setLoadingGroupList(false);
74     }
75   }
76 
77   async function refreshTeachers() {
78     setLoadingTeacherList(true);
79     try {
80       const res = await fetch("/api/teachers", {
81         method: "GET",
82         headers: { "Content-Type": "application/json" },
83         cache: "no-store",
84       });
85 
86       if (res.ok) {
87         const data = (await res.json()) as Teacher[];
88         setTeachers(
89           data.map((teacher) => ({
90             id: teacher.id,
91             name: teacher.name,
92             email: teacher.email,
93             isTeacher: teacher.isTeacher,
94             createdAt: teacher.createdAt,
95             _count: {
96               classes: teacher._count?.classes ?? 0,
97             },
98           }))
99         );
100       } else {
101         console.error("Erro ao carregar professores");
102       }
103     } catch (error) {
104       console.error("Erro ao buscar professores", error);
105     } finally {
106       setLoadingTeacherList(false);
107     }
108   }
109 
110   return (
111     <div className="mt-8 space-y-8 text-left">
112       <div>
113         <p className="text-sm text-slate-500">
114           {userName && userEmail ? (
115             <>
116               Logado como{" "}
117               <span className="font-medium">
118                 {userName} ({userEmail})
119               </span>
120               .
121             </>
122           ) : null}
123         </p>
124       </div>
125 
126       <section className="space-y-4">
127         <div className="flex items-start justify-between gap-4">
128           <div>
129             <h2 className="text-lg font-semibold text-slate-900">
130               Grupos de escolas
131             </h2>
132             <p className="mt-1 text-sm text-slate-500">
133               Organize suas escolas em grupos, como redes municipais, privadas ou
134               projetos específicos.
135             </p>
136           </div>
137 
138           <button
139             type="button"
140             onClick={() => setOpenGroupModal(true)}
141             className="inline-flex items-center rounded-md bg-sky-600 px-4 py-2 text-sm font-medium text-white shadow-sm hover:bg-sky-700 focus:outline-none focus:ring-2 focus:ring-sky-500 focus:ring-offset-2"
142           >
143             + Criar grupo
144           </button>
145         </div>
146 
147         <div className="rounded-lg border border-slate-200 bg-white p-4 shadow-sm">
148           {loadingGroupList && (
149             <p className="mb-3 text-sm text-slate-500">Atualizando grupos…</p>
150           )}
151 
152           {groups.length === 0 ? (
153             <p className="text-sm text-slate-500">
154               Você ainda não possui nenhum grupo de escolas. Clique em{" "}
155               <span className="font-semibold">“Criar grupo”</span> para começar.
156             </p>
157           ) : (
158             <ul className="space-y-3">
159               {groups.map((group) => (
160                 <li key={group.id}>
161                   <Link
162                     href={`/groups/${group.id}`}
163                     className="flex items-center justify-between rounded-md border border-slate-200 px-4 py-3 transition hover:bg-slate-50 focus:outline-none focus:ring-2 focus:ring-sky-500"
164                   >
165                     <div>
166                       <p className="text-sm font-medium text-slate-900">
167                         {group.name}
168                       </p>
169                       <p className="mt-1 text-xs text-slate-500">
170                         {group._count.schools === 0
171                           ? "Nenhuma escola cadastrada ainda"
172                           : group._count.schools === 1
173                           ? "1 escola cadastrada"
174                           : `${group._count.schools} escolas cadastradas`}
175                       </p>
176                     </div>
177 
178                     <span className="text-sm font-medium text-sky-700">
179                       Abrir
180                     </span>
181                   </Link>
182                 </li>
183               ))}
184             </ul>
185           )}
186         </div>
187       </section>
188 
189       <section className="space-y-4">
190         <div className="flex items-start justify-between gap-4">
191           <div>
192             <h2 className="text-lg font-semibold text-slate-900">Professores</h2>
193             <p className="mt-1 text-sm text-slate-500">
194               Cadastre professores com e-mail e senha temporária para acesso ao
195               sistema.
196             </p>
197           </div>
198 
199           <button
200             type="button"
201             onClick={() => setOpenTeacherModal(true)}
202             className="inline-flex items-center rounded-md bg-sky-600 px-4 py-2 text-sm font-medium text-white shadow-sm hover:bg-sky-700 focus:outline-none focus:ring-2 focus:ring-sky-500 focus:ring-offset-2"
203           >
204             + Adicionar professor
205           </button>
206         </div>
207 
208         <div className="rounded-lg border border-slate-200 bg-white p-4 shadow-sm">
209           {loadingTeacherList && (
210             <p className="mb-3 text-sm text-slate-500">
211               Atualizando professores…
212             </p>
213           )}
214 
215           {teachers.length === 0 ? (
216             <p className="text-sm text-slate-500">
217               Você ainda não cadastrou professores. Clique em{" "}
218               <span className="font-semibold">“Adicionar professor”</span> para
219               começar.
220             </p>
221           ) : (
222             <ul className="space-y-3">
223               {teachers.map((teacher) => (
224                 <li key={teacher.id}>
225                   <Link
226                     href={`/teachers/${teacher.id}`}
227                     className="flex items-center justify-between rounded-md border border-slate-200 px-4 py-3 transition hover:bg-slate-50 focus:outline-none focus:ring-2 focus:ring-sky-500"
228                   >
229                     <div>
230                       <p className="text-sm font-medium text-slate-900">
231                         {teacher.name}
232                       </p>
233                       <p className="mt-1 text-xs text-slate-500">{teacher.email}</p>
234                     </div>
235 
236                     <div className="text-right">
237                       <p className="text-sm font-medium text-slate-700">
238                         {teacher._count.classes === 0
239                           ? "Nenhuma turma vinculada"
240                           : teacher._count.classes === 1
241                           ? "1 turma vinculada"
242                           : `${teacher._count.classes} turmas vinculadas`}
243                       </p>
244                       <p className="mt-1 text-xs font-medium text-sky-700">
245                         Abrir
246                       </p>
247                     </div>
248                   </Link>
249                 </li>
250               ))}
251             </ul>
252           )}
253         </div>
254       </section>
255 
256       <CreateGroupModal
257         open={openGroupModal}
258         onClose={() => setOpenGroupModal(false)}
259         onCreated={refreshGroups}
260       />

263         open={openTeacherModal}
264         onClose={() => setOpenTeacherModal(false)}
265         onCreated={refreshTeachers}
266       />

269 }
