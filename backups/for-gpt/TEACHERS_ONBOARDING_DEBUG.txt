1 import { NextRequest, NextResponse } from "next/server";
2 import { prisma } from "@/lib/prisma";
3 import { hashPassword, createSession } from "@/lib/auth";
4 
5 export async function POST(req: NextRequest) {
6   const formData = await req.formData();
7   const name = String(formData.get("name") ?? "").trim();
8   const email = String(formData.get("email") ?? "").toLowerCase().trim();
9   const password = String(formData.get("password") ?? "");
10 
11   if (!name || !email || !password) {
12     const url = new URL("/register?error=missing_fields", req.url);
13     return NextResponse.redirect(url);
14   }
15 
16   const existing = await prisma.user.findUnique({
17     where: { email },
18   });
19 
20   if (existing) {
21     const url = new URL("/register?error=email_in_use", req.url);
22     return NextResponse.redirect(url);
23   }
24 
25   const passwordHash = await hashPassword(password);
26 
27   const user = await prisma.user.create({
28     data: {
29       name,
30       email,
31       passwordHash,
32     },
33   });
34 
35   const redirectUrl = await createSession(user.id, req.url);
36   return NextResponse.redirect(redirectUrl);
37 }

===== app/api/auth/login/route.ts =====
1 import { NextRequest, NextResponse } from "next/server";
2 import { prisma } from "@/lib/prisma";
3 import { verifyPassword, createSession } from "@/lib/auth";
4 
5 export async function POST(req: NextRequest) {
6   const formData = await req.formData();
7   const email = String(formData.get("email") ?? "").toLowerCase().trim();
8   const password = String(formData.get("password") ?? "");
9 
10   if (!email || !password) {
11     const url = new URL("/login?error=missing_fields", req.url);
12     return NextResponse.redirect(url);
13   }
14 
15   const user = await prisma.user.findUnique({
16     where: { email },
17   });
18 
19   if (!user) {
20     const url = new URL("/login?error=invalid_credentials", req.url);
21     return NextResponse.redirect(url);
22   }
23 
24   const valid = await verifyPassword(password, user.passwordHash);
25 
26   if (!valid) {
27     const url = new URL("/login?error=invalid_credentials", req.url);
28     return NextResponse.redirect(url);
29   }
30 
31   const redirectUrl = await createSession(user.id, req.url);
32   return NextResponse.redirect(redirectUrl);
33 }

===== lib/auth.ts =====
1 import { cookies } from "next/headers";
2 import { prisma } from "./prisma";
3 import { compare, hash } from "bcryptjs";
4 
5 const SESSION_COOKIE_NAME = "educc_session";
6 const SESSION_DAYS = 30;
7 
8 export async function hashPassword(password: string) {
9   return hash(password, 10);
10 }
11 
12 export async function verifyPassword(password: string, passwordHash: string) {
13   return compare(password, passwordHash);
14 }
15 
16 export async function createSession(userId: string, baseUrl: string) {
17   const expiresAt = new Date();
18   expiresAt.setDate(expiresAt.getDate() + SESSION_DAYS);
19 
20   const session = await prisma.session.create({
21     data: {
22       userId,
23       expiresAt,
24     },
25   });
26 
27   const cookieStore = await cookies();
28 
29   cookieStore.set(SESSION_COOKIE_NAME, session.id, {
30     httpOnly: true,
31     secure: true,
32     sameSite: "lax",
33     path: "/",
34     expires: expiresAt,
35   });
36 
37   return new URL("/dashboard", baseUrl);
38 }
39 
40 export async function destroySession() {
41   const cookieStore = await cookies();
42   const sessionId = cookieStore.get(SESSION_COOKIE_NAME)?.value;
43 
44   if (sessionId) {
45     await prisma.session.deleteMany({
46       where: { id: sessionId },
47     });
48 
49     cookieStore.set(SESSION_COOKIE_NAME, "", {
50       httpOnly: true,
51       secure: true,
52       sameSite: "lax",
53       path: "/",
54       maxAge: 0,
55     });
56   }
57 }
58 
59 export async function getSessionUser() {
60   const cookieStore = await cookies();
61   const sessionId = cookieStore.get(SESSION_COOKIE_NAME)?.value;
62 
63   if (!sessionId) return null;
64 
65   const session = await prisma.session.findUnique({
66     where: { id: sessionId },
67     include: { user: true },
68   });
69 
70   if (!session) return null;
71   if (session.expiresAt < new Date()) {
72     return null;
73   }
74 
75   return session.user;
76 }
