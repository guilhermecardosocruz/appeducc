===== app/api/groups/[groupId]/schools/route.ts =====
1 import { NextRequest, NextResponse } from "next/server";
2 import { getSessionUser } from "@/lib/auth";
3 import { prisma } from "@/lib/prisma";
4 
5 function isGroupManagerRole(role: string | null | undefined) {
6   const normalized = String(role ?? "").trim().toUpperCase();
7   return normalized === "OWNER" || normalized === "MANAGER";
8 }
9 
10 async function getGroupAccess(userId: string, groupId: string) {
11   const [groupMembership, schoolMemberships] = await Promise.all([
12     prisma.groupMember.findUnique({
13       where: {
14         userId_groupId: {
15           userId,
16           groupId,
17         },
18       },
19     }),
20     prisma.schoolMember.findMany({
21       where: {
22         userId,
23         school: {
24           groupId,
25         },
26       },
27       select: {
28         schoolId: true,
29         role: true,
30       },
31     }),
32   ]);
33 
34   const hasAccess = Boolean(groupMembership) || schoolMemberships.length > 0;
35   const canManage = Boolean(
36     groupMembership && isGroupManagerRole(groupMembership.role)
37   );
38 
39   return {
40     groupMembership,
41     schoolMemberships,
42     hasAccess,
43     canManage,
44   };
45 }
46 
47 export async function GET(
48   _req: NextRequest,
49   { params }: { params: Promise<{ groupId: string }> }
50 ) {
51   const user = await getSessionUser();
52   if (!user) {
53     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
54   }
55 
56   const { groupId } = await params;
57   const access = await getGroupAccess(user.id, groupId);
58 
59   if (!access.hasAccess) {
60     return NextResponse.json({ error: "Forbidden" }, { status: 403 });
61   }
62 
63   const schoolIds = access.groupMembership
64     ? undefined
65     : access.schoolMemberships.map((item) => item.schoolId);
66 
67   const schools = await prisma.school.findMany({
68     where: {
69       groupId,
70       ...(schoolIds ? { id: { in: schoolIds } } : {}),
71     },
72     include: {
73       _count: {
74         select: { classes: true },
75       },
76     },
77     orderBy: { name: "asc" },
78   });
79 
80   return NextResponse.json(schools);
81 }
82 
83 export async function POST(
84   req: NextRequest,
85   { params }: { params: Promise<{ groupId: string }> }
86 ) {
87   const user = await getSessionUser();
88   if (!user) {
89     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
90   }
91 
92   const { groupId } = await params;
93   const access = await getGroupAccess(user.id, groupId);
94 
95   if (!access.canManage) {
96     return NextResponse.json({ error: "Forbidden" }, { status: 403 });
97   }
98 
99   const data = await req.json();
100   const name = String(data.name ?? "").trim();
101 
102   if (!name) {
103     return NextResponse.json({ error: "Missing school name" }, { status: 400 });
104   }
105 
106   const school = await prisma.school.create({
107     data: {
108       name,
109       groupId,
110       members: {
111         create: {
112           userId: user.id,
113           role: "DIRECTOR",
114         },
115       },
116     },
117     include: {
118       _count: {
119         select: { classes: true },
120       },
121     },
122   });
123 
124   return NextResponse.json(school, { status: 201 });
125 }
