1 import { NextRequest, NextResponse } from "next/server";
2 import { getSessionUser } from "@/lib/auth";
3 import { prisma } from "@/lib/prisma";
4 
5 async function ensureGroupAccess(userId: string, groupId: string) {
6   const membership = await prisma.groupMember.findUnique({
7     where: {
8       userId_groupId: {
9         userId,
10         groupId,
11       },
12     },
13   });
14 
15   return membership;
16 }
17 
18 export async function GET(
19   _req: NextRequest,
20   { params }: { params: Promise<{ groupId: string }> }
21 ) {
22   const user = await getSessionUser();
23   if (!user) {
24     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
25   }
26 
27   const { groupId } = await params;
28   const membership = await ensureGroupAccess(user.id, groupId);
29 
30   if (!membership) {
31     return NextResponse.json({ error: "Forbidden" }, { status: 403 });
32   }
33 
34   const schools = await prisma.school.findMany({
35     where: { groupId },
36     include: {
37       _count: {
38         select: { classes: true },
39       },
40     },
41     orderBy: { createdAt: "desc" },
42   });
43 
44   return NextResponse.json(schools);
45 }
46 
47 export async function POST(
48   req: NextRequest,
49   { params }: { params: Promise<{ groupId: string }> }
50 ) {
51   const user = await getSessionUser();
52   if (!user) {
53     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
54   }
55 
56   const { groupId } = await params;
57   const membership = await ensureGroupAccess(user.id, groupId);
58 
59   if (!membership) {
60     return NextResponse.json({ error: "Forbidden" }, { status: 403 });
61   }
62 
63   const data = await req.json();
64   const name = String(data.name ?? "").trim();
65 
66   if (!name) {
67     return NextResponse.json({ error: "Missing school name" }, { status: 400 });
68   }
69 
70   const school = await prisma.school.create({
71     data: {
72       name,
73       groupId,
74       members: {
75         create: {
76           userId: user.id,
77           role: "DIRECTOR",
78         },
79       },
80     },
81     include: {
82       _count: {
83         select: { classes: true },
84       },
85     },
86   });
87 
88   return NextResponse.json(school, { status: 201 });
89 }
