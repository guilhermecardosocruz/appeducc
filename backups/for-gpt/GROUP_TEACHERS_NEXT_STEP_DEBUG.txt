81 export async function POST(
82   req: NextRequest,
83   { params }: { params: Promise<{ teacherId: string }> }
84 ) {
85   const user = await getSessionUser();
86 
87   if (!user) {
88     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
89   }
90 
91   const { teacherId } = await params;
92   const teacher = await getTeacherOwnedByUser(user.id, teacherId);
93 
94   if (!teacher) {
95     return NextResponse.json({ error: "Teacher not found" }, { status: 404 });
96   }
97 
98   const data = await req.json();
99   const classId = String(data.classId ?? "").trim();
100 
101   if (!classId) {
102     return NextResponse.json({ error: "Missing classId" }, { status: 400 });
103   }
104 
105   const targetClass = await prisma.class.findFirst({
106     where: {
107       id: classId,
108       school: {
109         members: {
110           some: {
111             userId: user.id,
112           },
113         },
114       },
115     },
116     include: {
117       school: {
118         include: {
119           group: true,
120         },
121       },
122     },
123   });
124 
125   if (!targetClass) {
126     return NextResponse.json({ error: "Class not found" }, { status: 404 });
127   }
128 
129   await prisma.$transaction(async (tx) => {
130     await tx.class.update({
131       where: { id: classId },
132       data: {
133         teacherId: teacher.id,
134       },
135     });
136 
137     await tx.schoolMember.upsert({
138       where: {
139         userId_schoolId: {
140           userId: teacher.id,
141           schoolId: targetClass.schoolId,
142         },
143       },
144       update: {
145         role: "TEACHER",
146       },
147       create: {
148         userId: teacher.id,
149         schoolId: targetClass.schoolId,
150         role: "TEACHER",
151       },
152     });
153   });
154 
155   const payload = await getTeacherClassesPayload(user.id, teacher.id);
156   return NextResponse.json(payload);
157 }

159 export async function DELETE(
160   req: NextRequest,
161   { params }: { params: Promise<{ teacherId: string }> }

163   const user = await getSessionUser();
164 
165   if (!user) {
166     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

172   if (!teacher) {

176   const data = await req.json();

181   }

183   const targetClass = await prisma.class.findFirst({
184     where: {

202     },
203   });
204 
205   if (!targetClass) {
206     return NextResponse.json({ error: "Class not found" }, { status: 404 });
207   }
208 
209   await prisma.$transaction(async (tx) => {
210     await tx.class.update({
211       where: { id: classId },
212       data: {
213         teacherId: null,
214       },
215     });
216 
217     const remainingTeacherClassesInSchool = await tx.class.count({
218       where: {
219         teacherId: teacher.id,
220         schoolId: targetClass.schoolId,
221       },
222     });
223 
224     if (remainingTeacherClassesInSchool === 0) {
225       await tx.schoolMember.deleteMany({
226         where: {
227           userId: teacher.id,
228           schoolId: targetClass.schoolId,
229           role: "TEACHER",
230         },
231       });
232     }
233   });
234

236   return NextResponse.json(payload);
