1 import { NextResponse } from "next/server";
2 import { prisma } from "@/lib/prisma";
3 import { getSessionUser } from "@/lib/auth";
4 
5 type Params = {
6   params: Promise<{ groupId: string }>;
7 };
8 
9 export async function GET(req: Request, { params }: Params) {
10   const user = await getSessionUser();
11   if (!user) {
12     return NextResponse.json({ error: "Não autenticado" }, { status: 401 });
13   }
14 
15   const { groupId } = await params;
16 
17   const schools = await prisma.school.findMany({
18     where: {
19       groupId,
20     },
21     include: {
22       classes: {
23         include: {
24           students: true,
25           attendances: {
26             include: {
27               presences: true,
28             },
29           },
30         },
31       },
32     },
33     orderBy: { name: "asc" },
34   });
35 
36   const report = schools.map((school) => {
37     let totalStudents = 0;
38     let totalPresences = 0;
39     let totalAbsences = 0;
40 
41     school.classes.forEach((cls) => {
42       totalStudents += cls.students.length;
43 
44       cls.attendances.forEach((attendance) => {
45         attendance.presences.forEach((presence) => {
46           if (presence.present) {
47             totalPresences++;
48           } else {
49             totalAbsences++;
50           }
51         });
52       });
53     });
54 
55     const total = totalPresences + totalAbsences;
56     const presenceRate = total > 0 ? (totalPresences / total) * 100 : 0;
57 
58     return {
59       schoolId: school.id,
60       schoolName: school.name,
61       classes: school.classes.length,
62       students: totalStudents,
63       presences: totalPresences,
64       absences: totalAbsences,
65       presenceRate: Number(presenceRate.toFixed(2)),
66     };
67   });
68 
69   return NextResponse.json(report);
70 }
