9 function generateTeacherPassword(name: string, cpf: string) {
10   const cleanName = name.trim().replace(/\s+/g, "");
11   const cleanCpf = normalizeCpf(cpf);
12 
13   if (cleanName.length < 3 || cleanCpf.length < 6) {
14     return null;
15   }
16 
17   const first = cleanName[0].toUpperCase();
18   const second = cleanName[1].toLowerCase();
19   const third = cleanName[2].toUpperCase();
20   const cpfPart = cleanCpf.slice(0, 6);
21 
22   return `${first}${second}${third}@${cpfPart}.`;
23 }

62 export async function POST(req: NextRequest) {
63   const user = await getSessionUser();
64 
65   if (!user) {
66     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
67   }
68 
69   const data = await req.json();
70   const name = String(data.name ?? "").trim();
71   const email = String(data.email ?? "").toLowerCase().trim();
72   const cpfRaw = String(data.cpf ?? "").trim();
73   const cpf = normalizeCpf(cpfRaw);
74 
75   if (!name || !email || !cpf) {
76     return NextResponse.json(
77       { error: "Missing teacher name, email or CPF" },
78       { status: 400 }
79     );
80   }
81 
82   if (cpf.length !== 11) {
83     return NextResponse.json(
84       { error: "CPF must contain 11 digits" },
85       { status: 400 }
86     );
87   }
88 
89   const existingByEmail = await prisma.user.findUnique({
90     where: { email },
91   });
92 
93   // Se usuário já existe → não altera senha
94   if (existingByEmail) {
95     const updated = await prisma.user.update({
96       where: { id: existingByEmail.id },
97       data: {
98         isTeacher: true,
99         createdById: user.id,
100         cpf: existingByEmail.cpf ?? cpf,
101       },
102       include: {
103         _count: {
104           select: {
105             classes: true,
106           },
107         },
108       },
109     });
110 
111     return NextResponse.json(
112       {
113         teacher: {
114           id: updated.id,
115           name: updated.name,
116           email: updated.email,
117           cpf: updated.cpf,
118           isTeacher: updated.isTeacher,
119           createdAt: updated.createdAt,
120           _count: {
121             classes: updated._count.classes,
122           },
123         },
124         message: "Usuário já existia. Senha não foi alterada.",
125       },
126       { status: 200 }
127     );
128   }
129 
130   const temporaryPassword = generateTeacherPassword(name, cpf);
131 
132   if (!temporaryPassword) {
133     return NextResponse.json(
134       { error: "Could not generate password from provided name and CPF" },
135       { status: 400 }
136     );
137   }
138 
139   const passwordHash = await hashPassword(temporaryPassword);
140 
141   const teacher = await prisma.user.create({
142     data: {
143       name,
144       email,
145       cpf,
146       passwordHash,
147       isTeacher: true,
148       createdById: user.id,
149     },
150     include: {
151       _count: {
152         select: {
153           classes: true,
154         },
155       },
156     },
157   });
158 
159   return NextResponse.json(
160     {
161       teacher: {
162         id: teacher.id,
163         name: teacher.name,
164         email: teacher.email,
165         cpf: teacher.cpf,
166         isTeacher: teacher.isTeacher,
167         createdAt: teacher.createdAt,
168         _count: {
169           classes: teacher._count.classes,
170         },
171       },
172       temporaryPassword,
173     },
174     { status: 201 }
175   );
176 }

===== app/api/auth/register/route.ts =====
1 import { NextRequest, NextResponse } from "next/server";
2 import { prisma } from "@/lib/prisma";
3 import { hashPassword, createSession } from "@/lib/auth";
4 
5 function normalizeCpf(cpf: string) {
6   return cpf.replace(/\D/g, "");
7 }
8 
9 export async function POST(req: NextRequest) {
10   const formData = await req.formData();
11   const name = String(formData.get("name") ?? "").trim();
12   const email = String(formData.get("email") ?? "").toLowerCase().trim();
13   const cpf = normalizeCpf(String(formData.get("cpf") ?? "").trim());
14   const password = String(formData.get("password") ?? "");
15 
16   if (!name || !email || !cpf || !password) {
17     const url = new URL("/register?error=missing_fields", req.url);
18     return NextResponse.redirect(url);
19   }
20 
21   if (cpf.length !== 11) {
22     const url = new URL("/register?error=invalid_cpf", req.url);
23     return NextResponse.redirect(url);
24   }
25 
26   const existingByEmail = await prisma.user.findUnique({
27     where: { email },
28   });
29 
30   if (existingByEmail) {
31     const url = new URL("/register?error=email_in_use", req.url);
32     return NextResponse.redirect(url);
33   }
34 
35   const existingByCpf = await prisma.user.findFirst({
36     where: { cpf },
37   });
38 
39   if (existingByCpf) {
40     const url = new URL("/register?error=cpf_in_use", req.url);
41     return NextResponse.redirect(url);
42   }
43 
44   const passwordHash = await hashPassword(password);
45 
46   const user = await prisma.user.create({
47     data: {
48       name,
49       email,
50       cpf,
51       passwordHash,
52     },
53   });
54 
55   const redirectUrl = await createSession(user.id, req.url);
56   return NextResponse.redirect(redirectUrl);
57 }
