===== app/api/classes/[classId]/students/[studentId]/route.ts =====
1 import { NextRequest, NextResponse } from "next/server";
2 import { getSessionUser } from "@/lib/auth";
3 import { prisma } from "@/lib/prisma";
4 
5 async function ensureClassAccess(userId: string, classId: string) {
6   const foundClass = await prisma.class.findUnique({
7     where: { id: classId },
8     include: { school: true },
9   });
10 
11   if (!foundClass) return null;
12 
13   const membership = await prisma.schoolMember.findUnique({
14     where: {
15       userId_schoolId: {
16         userId,
17         schoolId: foundClass.schoolId,
18       },
19     },
20   });
21 
22   if (!membership) return null;
23 
24   return foundClass;
25 }
26 
27 export async function GET(
28   req: NextRequest,
29   { params }: { params: Promise<{ classId: string; studentId: string }> }
30 ) {
31   const user = await getSessionUser();
32   if (!user) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
33 
34   const { classId, studentId } = await params;
35 
36   const foundClass = await ensureClassAccess(user.id, classId);
37   if (!foundClass) return NextResponse.json({ error: "Forbidden" }, { status: 403 });
38 
39   const student = await prisma.student.findUnique({
40     where: { id: studentId },
41     include: {
42       presences: true,
43     },
44   });
45 
46   if (!student || student.classId !== classId) {
47     return NextResponse.json({ error: "Student not found" }, { status: 404 });
48   }
49 
50   const total = student.presences.length;
51   const presents = student.presences.filter(p => p.present).length;
52   const absents = total - presents;
53   const percentage = total ? Math.round((presents / total) * 100) : 0;
54 
55   return NextResponse.json({
56     id: student.id,
57     name: student.name,
58     deletedAt: student.deletedAt,
59     deletedReason: student.deletedReason,
60     stats: {
61       total,
62       presents,
63       absents,
64       percentage,
65     },
66   });
67 }
68 
69 export async function PATCH(
70   req: NextRequest,
71   { params }: { params: Promise<{ classId: string; studentId: string }> }
72 ) {
73   const user = await getSessionUser();
74   if (!user) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
75 
76   const { classId, studentId } = await params;
77   const { name } = await req.json();
78 
79   const foundClass = await ensureClassAccess(user.id, classId);
80   if (!foundClass) return NextResponse.json({ error: "Forbidden" }, { status: 403 });
81 
82   await prisma.student.update({
83     where: { id: studentId },
84     data: { name },
85   });
86 
87   return NextResponse.json({ success: true });
88 }
