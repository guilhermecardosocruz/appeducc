1 import { notFound } from "next/navigation";
2 import StudentsManagerClient from "@/components/StudentsManagerClient";
3 import { getSessionUser } from "@/lib/auth";
4 import { prisma } from "@/lib/prisma";
5 
6 type PageProps = {
7   params: Promise<{ classId: string }>;
8 };
9 
10 function isManagerRole(role: string | null | undefined) {
11   const normalized = String(role ?? "").trim().toUpperCase();
12   return ["OWNER", "MANAGER"].includes(normalized);
13 }
14 
15 export default async function ClassStudentsPage({ params }: PageProps) {
16   const user = await getSessionUser();
17 
18   if (!user) {
19     notFound();
20   }
21 
22   const { classId } = await params;
23 
24   const foundClass = await prisma.class.findUnique({
25     where: { id: classId },
26     include: {
27       school: true,
28       students: {
29         orderBy: { name: "asc" },
30       },
31     },
32   });
33 
34   if (!foundClass) {
35     notFound();
36   }
37 
38   const [schoolMembership, groupMembership] = await Promise.all([
39     prisma.schoolMember.findUnique({
40       where: {
41         userId_schoolId: {
42           userId: user.id,
43           schoolId: foundClass.schoolId,
44         },
45       },
46     }),
47     prisma.groupMember.findUnique({
48       where: {
49         userId_groupId: {
50           userId: user.id,
51           groupId: foundClass.school.groupId,
52         },
53       },
54     }),
55   ]);
56 
57   const hasAccess = Boolean(schoolMembership) || Boolean(groupMembership);
58   const canManageStudents =
59     Boolean(schoolMembership) ||
60     Boolean(groupMembership && isManagerRole(groupMembership.role));
61 
62   if (!hasAccess) {
63     notFound();
64   }
65 
66   const students = foundClass.students.map((student) => ({
67     id: student.id,
68     name: student.name,
69     status: student.status as "ACTIVE" | "PENDING_DELETE" | "DELETED",
70     createdAt: student.createdAt.toISOString(),
71     deletedAt: student.deletedAt?.toISOString() ?? null,
72     deletedReason: student.deletedReason ?? null,
73   }));
74 
75   return (
76     <StudentsManagerClient
77       classId={foundClass.id}
78       className={foundClass.name}
79       canImportSpreadsheet={!user.isTeacher && canManageStudents}
80       canManageStudents={canManageStudents}
81       initialStudents={students}
82     />
83   );
84 }
