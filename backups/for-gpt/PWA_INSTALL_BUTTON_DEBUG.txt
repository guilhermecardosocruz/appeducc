26     standalone?: boolean;
27   };

35 export default function PwaInstallButton() {
36   const [deferredPrompt, setDeferredPrompt] =
37     useState<BeforeInstallPromptEvent | null>(null);
38   const [installed, setInstalled] = useState<boolean>(() =>
39     isInStandaloneMode()
40   );
41   const [showIosHelp, setShowIosHelp] = useState(false);
42   const [isMobileDevice] = useState<boolean>(() => isMobileOrTablet());
43 
44   useEffect(() => {
45     if (!isMobileDevice) return;
46 
47     if ("serviceWorker" in navigator) {
48       void navigator.serviceWorker.register("/sw.js").catch((error) => {
49         console.error("Erro ao registrar service worker", error);
50       });
51     }
52 
53     function handleBeforeInstallPrompt(event: Event) {
54       event.preventDefault();
55       setDeferredPrompt(event as BeforeInstallPromptEvent);
56     }
57 
58     function handleAppInstalled() {
59       setInstalled(true);
60       setDeferredPrompt(null);
61       setShowIosHelp(false);
62     }
63 
64     window.addEventListener(
65       "beforeinstallprompt",
66       handleBeforeInstallPrompt as EventListener
67     );
68     window.addEventListener("appinstalled", handleAppInstalled);
69 
70     return () => {
71       window.removeEventListener(
72         "beforeinstallprompt",
73         handleBeforeInstallPrompt as EventListener
74       );
75       window.removeEventListener("appinstalled", handleAppInstalled);
76     };
77   }, [isMobileDevice]);
78 
79   const canUsePrompt = useMemo(
80     () => !installed && Boolean(deferredPrompt),
81     [installed, deferredPrompt]
82   );
83 
84   async function handleInstall() {
85     if (installed || !isMobileDevice) return;
86 
87     if (deferredPrompt) {
88       await deferredPrompt.prompt();
89       await deferredPrompt.userChoice;
90       setDeferredPrompt(null);
91       return;
92     }
93 
94     if (isIos()) {
95       setShowIosHelp((current) => !current);
96       return;
97     }
98 
99     alert(
100       "A instalação automática ainda não está disponível neste navegador. No menu do navegador, procure por 'Instalar app' ou 'Adicionar à tela inicial'."
101     );
102   }
103 
104   if (!isMobileDevice) {
105     return null;
106   }
107 
108   if (installed) {
109     return (
110       <div className="rounded-xl border border-emerald-200 bg-emerald-50 px-4 py-3 text-sm font-medium text-emerald-700">
111         App já instalado neste dispositivo.
112       </div>
113     );
114   }
115 
116   return (
117     <div className="space-y-3">
118       <button
119         type="button"
120         onClick={() => void handleInstall()}
121         className="inline-flex w-full items-center justify-center rounded-xl border border-sky-200 bg-sky-50 px-4 py-2.5 text-sm font-semibold text-sky-700 transition hover:bg-sky-100"
122       >
123         {canUsePrompt ? "Instalar no dispositivo" : "Instalar no dispositivo"}
124       </button>
125 
126       {showIosHelp && (
127         <div className="rounded-xl border border-slate-200 bg-slate-50 px-4 py-3 text-sm text-slate-600">
128           No iPhone/iPad, toque em <span className="font-semibold">Compartilhar</span> no Safari e depois em{" "}
129           <span className="font-semibold">Adicionar à Tela de Início</span>.
130         </div>
131       )}
132     </div>
133   );
134 }
