1 import { NextRequest, NextResponse } from "next/server";
2 import { getSessionUser } from "@/lib/auth";

5 async function ensureClassAccess(userId: string, classId: string) {
6   const foundClass = await prisma.class.findUnique({
7     where: { id: classId },
8     include: {
9       school: true,
10     },
11   });
12 
13   if (!foundClass) return null;
14 
15   const membership = await prisma.schoolMember.findUnique({
16     where: {
17       userId_schoolId: {
18         userId,
19         schoolId: foundClass.schoolId,
20       },
21     },
22   });
23 
24   if (!membership) return null;
25 
26   return foundClass;
27 }

29 export async function GET(
30   _req: NextRequest,
31   { params }: { params: Promise<{ classId: string }> }
32 ) {
33   const user = await getSessionUser();
34 
35   if (!user) {
36     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
37   }
38 
39   const { classId } = await params;
40   const foundClass = await ensureClassAccess(user.id, classId);
41 
42   if (!foundClass) {
43     return NextResponse.json({ error: "Forbidden" }, { status: 403 });
44   }
45 
46   const students = await prisma.student.findMany({
47     where: { classId },
48     orderBy: { name: "asc" },
49   });
50 
51   return NextResponse.json(students);
52 }

54 export async function POST(
55   req: NextRequest,
56   { params }: { params: Promise<{ classId: string }> }
57 ) {
58   const user = await getSessionUser();
59 
60   if (!user) {
61     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
62   }
63 
64   const { classId } = await params;
65   const foundClass = await ensureClassAccess(user.id, classId);
66 
67   if (!foundClass) {
68     return NextResponse.json({ error: "Forbidden" }, { status: 403 });
69   }
70 
71   const data = await req.json();
72   const name = String(data.name ?? "").trim();
73   const attendanceId = String(data.attendanceId ?? "").trim();
74 
75   if (!name) {
76     return NextResponse.json({ error: "Missing student name" }, { status: 400 });
77   }
78 
79   const existingStudent = await prisma.student.findFirst({
80     where: {
81       classId,
82       name,
83     },
84   });
85 
86   if (existingStudent) {
87     return NextResponse.json(
88       { error: "Student already exists in this class" },
89       { status: 409 }
90     );
91   }
92 
93   const createdStudent = await prisma.student.create({
94     data: {
95       name,
96       classId,
97     },
98   });
99 
100   const classAttendances = await prisma.attendance.findMany({
101     where: { classId },
102     select: { id: true },
103   });
104 
105   if (classAttendances.length > 0) {
106     await prisma.attendancePresence.createMany({
107       data: classAttendances.map((attendance) => ({
108         attendanceId: attendance.id,
109         studentId: createdStudent.id,
110         present: false,
111       })),
112       skipDuplicates: true,
113     });
114   }
115 
116   let createdPresence = null;
117 
118   if (attendanceId) {
119     const attendance = await prisma.attendance.findUnique({
120       where: { id: attendanceId },
121     });
122 
123     if (!attendance || attendance.classId !== classId) {
124       return NextResponse.json(
125         { error: "Attendance not found for this class" },
126         { status: 400 }
127       );
128     }
129 
130     createdPresence = await prisma.attendancePresence.findFirst({
131       where: {
132         attendanceId,
133         studentId: createdStudent.id,
134       },
135       include: {
136         student: {
137           select: {
138             id: true,
139             name: true,
140           },
141         },
142       },
143     });
144   }
145 
146   return NextResponse.json(
147     {
148       student: createdStudent,
149       presence: createdPresence,
150     },
151     { status: 201 }
152   );
153 }
