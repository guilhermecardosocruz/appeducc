5 //
6 // GET /api/groups
7 // Lista todos os grupos onde o usuário é membro
8 //
9 export async function GET() {
10   const user = await getSessionUser();
11   if (!user) {
12     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
13   }
14 
15   const groups = await prisma.group.findMany({
16     where: {
17       members: {
18         some: {
19           userId: user.id,
20         },
21       },
22     },
23     include: {
24       _count: { select: { schools: true } },
25     },
26     orderBy: { createdAt: "desc" },
27   });
28 
29   return NextResponse.json(groups);
30 }
31 
32 //
33 // POST /api/groups
34 // Cria um grupo e adiciona o usuário como OWNER
35 //
36 export async function POST(req: NextRequest) {
37   const user = await getSessionUser();
38   if (!user) {
39     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
40   }
41 
42   const data = await req.json();
43   const name = String(data.name ?? "").trim();
44 
45   if (!name) {
46     return NextResponse.json({ error: "Missing group name" }, { status: 400 });
47   }
48 
49   const group = await prisma.group.create({
50     data: {
51       name,
52       members: {
53         create: {
54           userId: user.id,
55           role: "OWNER",
56         },
57       },
58     },
59   });
60 
61   return NextResponse.json(group, { status: 201 });
62 }
