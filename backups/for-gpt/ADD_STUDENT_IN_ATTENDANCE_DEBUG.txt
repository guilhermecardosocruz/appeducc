36       studentName: student.name,

47 
48   function togglePresence(studentId: string) {
49     setPresences((current) =>
50       current.map((item) =>
51         item.studentId === studentId ? { ...item, present: !item.present } : item
52       )
53     );
54   }
55 
56   function markAll(value: boolean) {
57     setPresences((current) =>
58       current.map((item) => ({
59         ...item,
60         present: value,

76           title,
77           lessonDate,
78           presences: presences.map((item) => ({

94       console.error("Erro ao criar chamada", error);

96       setLoading(false);
