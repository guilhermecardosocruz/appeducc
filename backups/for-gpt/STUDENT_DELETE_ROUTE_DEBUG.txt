61 export async function POST(
62   req: NextRequest,
63   { params }: { params: Promise<{ classId: string }> }
64 ) {
65   const user = await getSessionUser();
66 
67   if (!user) {
68     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
69   }
70 
71   const { classId } = await params;
72   const foundClass = await ensureClassAccess(user.id, classId);
73 
74   if (!foundClass) {
75     return NextResponse.json({ error: "Forbidden" }, { status: 403 });
76   }
77 
78   const data = await req.json();
79   const name = String(data.name ?? "").trim();
80   const attendanceId = String(data.attendanceId ?? "").trim();
81 
82   if (!name) {
83     return NextResponse.json({ error: "Missing student name" }, { status: 400 });
84   }
85 
86   const existingStudent = await prisma.student.findFirst({
87     where: {
88       classId,
89       name,
90     },
91   });
92 
93   if (existingStudent) {
94     return NextResponse.json(
95       { error: "Student already exists in this class" },
96       { status: 409 }
97     );
98   }
99 
100   const createdStudent = await prisma.student.create({
101     data: {
102       name,
103       classId,
104     },
105   });
106 
107   const classAttendances = await prisma.attendance.findMany({
108     where: { classId },
109     select: { id: true },
110   });
111 
112   if (classAttendances.length > 0) {
113     await prisma.attendancePresence.createMany({
114       data: classAttendances.map((attendance) => ({
115         attendanceId: attendance.id,
116         studentId: createdStudent.id,
117         present: false,
118       })),
119       skipDuplicates: true,
120     });
121   }
122 
123   let createdPresence = null;
124 
125   if (attendanceId) {
126     const attendance = await prisma.attendance.findUnique({
127       where: { id: attendanceId },
128     });
129 
130     if (!attendance || attendance.classId !== classId) {
131       return NextResponse.json(
132         { error: "Attendance not found for this class" },
133         { status: 400 }
134       );
135     }
136 
137     createdPresence = await prisma.attendancePresence.findFirst({
138       where: {
139         attendanceId,
140         studentId: createdStudent.id,
141       },
142       include: {
143         student: {
144           select: {
145             id: true,
146             name: true,
147           },
148         },
149       },
150     });
151   }
152 
153   return NextResponse.json(
154     {
155       student: createdStudent,
156       presence: createdPresence,
157     },
158     { status: 201 }
159   );
160 }

27 export async function GET(
28   req: NextRequest,
29   { params }: { params: Promise<{ classId: string; studentId: string }> }
30 ) {
31   const user = await getSessionUser();
32   if (!user) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
33 
34   const { classId, studentId } = await params;
35 
36   const foundClass = await ensureClassAccess(user.id, classId);
37   if (!foundClass) return NextResponse.json({ error: "Forbidden" }, { status: 403 });
38 
39   const student = await prisma.student.findUnique({
40     where: { id: studentId },
41     include: {
42       presences: true,
43     },
44   });
45 
46   if (!student || student.classId !== classId) {
47     return NextResponse.json({ error: "Student not found" }, { status: 404 });
48   }
49 
50   const total = student.presences.length;
51   const presents = student.presences.filter(p => p.present).length;
52   const absents = total - presents;
53   const percentage = total ? Math.round((presents / total) * 100) : 0;
54 
55   return NextResponse.json({
56     id: student.id,
57     name: student.name,
58     deletedAt: student.deletedAt,
59     deletedReason: student.deletedReason,
60     stats: {
61       total,
62       presents,
63       absents,
64       percentage,
65     },
66   });
67 }
