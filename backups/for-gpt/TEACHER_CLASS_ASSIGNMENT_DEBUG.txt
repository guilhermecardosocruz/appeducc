1 import { NextRequest, NextResponse } from "next/server";
2 import { getSessionUser } from "@/lib/auth";
3 import { prisma } from "@/lib/prisma";
4 
5 async function ensureSchoolAccess(userId: string, schoolId: string) {
6   const membership = await prisma.schoolMember.findUnique({
7     where: {
8       userId_schoolId: {
9         userId,
10         schoolId,
11       },
12     },
13     include: {
14       school: true,
15     },
16   });
17 
18   return membership;
19 }
20 
21 export async function GET(
22   _req: NextRequest,
23   { params }: { params: Promise<{ schoolId: string }> }
24 ) {
25   const user = await getSessionUser();
26   if (!user) {
27     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
28   }
29 
30   const { schoolId } = await params;
31   const membership = await ensureSchoolAccess(user.id, schoolId);
32 
33   if (!membership) {
34     return NextResponse.json({ error: "Forbidden" }, { status: 403 });
35   }
36 
37   const classes = await prisma.class.findMany({
38     where: { schoolId },
39     include: {
40       _count: {
41         select: { students: true },
42       },
43     },
44     orderBy: { createdAt: "desc" },
45   });
46 
47   return NextResponse.json(classes);
48 }
49 
50 export async function POST(
51   req: NextRequest,
52   { params }: { params: Promise<{ schoolId: string }> }
53 ) {
54   const user = await getSessionUser();
55   if (!user) {
56     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
57   }
58 
59   const { schoolId } = await params;
60   const membership = await ensureSchoolAccess(user.id, schoolId);
61 
62   if (!membership) {
63     return NextResponse.json({ error: "Forbidden" }, { status: 403 });
64   }
65 
66   const data = await req.json();
67   const name = String(data.name ?? "").trim();
68   const yearRaw = data.year;
69 
70   if (!name) {
71     return NextResponse.json({ error: "Missing class name" }, { status: 400 });
72   }
73 
74   const year =
75     typeof yearRaw === "number" && Number.isFinite(yearRaw) ? yearRaw : null;
76 
77   const createdClass = await prisma.class.create({
78     data: {
79       name,
80       year,
81       schoolId,
82     },
83     include: {
84       _count: {
85         select: { students: true },
86       },
87     },
88   });
89 
90   return NextResponse.json(createdClass, { status: 201 });
91 }

5 function generateTemporaryPassword(length = 10) {
6   const chars = "ABCDEFGHJKLMNPQRSTUVWXYZabcdefghijkmnopqrstuvwxyz23456789";
7   let password = "";
8 
9   for (let i = 0; i < length; i += 1) {
10     password += chars[Math.floor(Math.random() * chars.length)];
11   }
12 
13   return password;
14 }

16 export async function GET() {

41 export async function POST(req: NextRequest) {
42   const user = await getSessionUser();
43 
44   if (!user) {
45     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
46   }
47 
48   const data = await req.json();
49   const name = String(data.name ?? "").trim();
50   const email = String(data.email ?? "").toLowerCase().trim();
51 
52   if (!name || !email) {
53     return NextResponse.json(
54       { error: "Missing teacher name or email" },
55       { status: 400 }
56     );
57   }
58 
59   const existing = await prisma.user.findUnique({
60     where: { email },
61   });
62 
63   if (existing) {
64     return NextResponse.json(
65       { error: "Email already in use" },
66       { status: 409 }
67     );
68   }
69 
70   const temporaryPassword = generateTemporaryPassword();
71   const passwordHash = await hashPassword(temporaryPassword);
72 
73   const teacher = await prisma.user.create({
74     data: {
75       name,
76       email,
77       passwordHash,
78       isTeacher: true,
79       createdById: user.id,
80     },
81     include: {
82       _count: {
83         select: {
84           classes: true,
85         },
86       },
87     },
88   });
89 
90   return NextResponse.json(
91     {
92       teacher: {
93         id: teacher.id,
94         name: teacher.name,
95         email: teacher.email,
96         isTeacher: teacher.isTeacher,
97         createdAt: teacher.createdAt,
98         _count: {
99           classes: teacher._count.classes,
100         },
101       },
102       temporaryPassword,
103     },
104     { status: 201 }
105   );
