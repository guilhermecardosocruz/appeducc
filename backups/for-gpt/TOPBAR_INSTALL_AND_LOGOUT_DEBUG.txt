5 type BeforeInstallPromptEvent = Event & {

10 function isIos() {
11   if (typeof window === "undefined") return false;
12   return /iphone|ipad|ipod/i.test(window.navigator.userAgent);
13 }
14 
15 function isInStandaloneMode() {
16   if (typeof window === "undefined") return false;
17 
18   const navigatorWithStandalone = window.navigator as Navigator & {
19     standalone?: boolean;
20   };
21 
22   return (
23     window.matchMedia("(display-mode: standalone)").matches ||
24     Boolean(navigatorWithStandalone.standalone)
25   );
26 }

28 export default function PwaInstallButton() {
29   const [deferredPrompt, setDeferredPrompt] =
30     useState<BeforeInstallPromptEvent | null>(null);
31   const [installed, setInstalled] = useState<boolean>(() =>
32     isInStandaloneMode()
33   );
34   const [showIosHelp, setShowIosHelp] = useState(false);
35 
36   useEffect(() => {
37     if ("serviceWorker" in navigator) {
38       void navigator.serviceWorker.register("/sw.js").catch((error) => {
39         console.error("Erro ao registrar service worker", error);
40       });
41     }
42 
43     function handleBeforeInstallPrompt(event: Event) {
44       event.preventDefault();
45       setDeferredPrompt(event as BeforeInstallPromptEvent);
46     }
47 
48     function handleAppInstalled() {
49       setInstalled(true);
50       setDeferredPrompt(null);
51       setShowIosHelp(false);
52     }
53 
54     window.addEventListener(
55       "beforeinstallprompt",
56       handleBeforeInstallPrompt as EventListener
57     );
58     window.addEventListener("appinstalled", handleAppInstalled);
59 
60     return () => {
61       window.removeEventListener(
62         "beforeinstallprompt",
63         handleBeforeInstallPrompt as EventListener
64       );
65       window.removeEventListener("appinstalled", handleAppInstalled);
66     };
67   }, []);
68 
69   const canUsePrompt = useMemo(
70     () => !installed && Boolean(deferredPrompt),
71     [installed, deferredPrompt]
72   );
73 
74   async function handleInstall() {
75     if (installed) return;
76 
77     if (deferredPrompt) {
78       await deferredPrompt.prompt();
79       await deferredPrompt.userChoice;
80       setDeferredPrompt(null);
81       return;
82     }
83 
84     if (isIos()) {
85       setShowIosHelp((current) => !current);
86       return;
87     }
88 
89     alert(
90       "A instalação automática ainda não está disponível neste navegador. No menu do navegador, procure por 'Instalar app' ou 'Adicionar à tela inicial'."
91     );
92   }
93 
94   if (installed) {
95     return (
96       <div className="rounded-xl border border-emerald-200 bg-emerald-50 px-4 py-3 text-sm font-medium text-emerald-700">
97         App já instalado neste dispositivo.
98       </div>
99     );
100   }
101 
102   return (
103     <div className="space-y-3">
104       <button
105         type="button"
106         onClick={() => void handleInstall()}
107         className="inline-flex w-full items-center justify-center rounded-xl border border-sky-200 bg-sky-50 px-4 py-2.5 text-sm font-semibold text-sky-700 transition hover:bg-sky-100"
108       >
109         {canUsePrompt ? "Instalar no celular" : "Instalar no celular"}
110       </button>
111 
112       {showIosHelp && (
113         <div className="rounded-xl border border-slate-200 bg-slate-50 px-4 py-3 text-sm text-slate-600">
114           No iPhone/iPad, toque em <span className="font-semibold">Compartilhar</span> no Safari e depois em{" "}
115           <span className="font-semibold">Adicionar à Tela de Início</span>.
116         </div>
117       )}
118     </div>
