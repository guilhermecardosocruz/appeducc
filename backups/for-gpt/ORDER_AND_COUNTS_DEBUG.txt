116       classId,
117       name,

126   }

129     data: {
130       name,
131       classId,
132       status: "ACTIVE",
133     },
134   });
135 
136   const classAttendances = await prisma.attendance.findMany({
137     where: { classId },
138     select: { id: true },
139   });
140 
141   if (classAttendances.length > 0) {
142     await prisma.attendancePresence.createMany({
143       data: classAttendances.map((attendance) => ({
144         attendanceId: attendance.id,
145         studentId: createdStudent.id,
146         present: false,
147       })),
148       skipDuplicates: true,
149     });
150   }
151 
152   let createdPresence = null;
153 
154   if (attendanceId) {
155     const attendance = await prisma.attendance.findUnique({
156       where: { id: attendanceId },
157     });
158 
159     if (!attendance || attendance.classId !== classId) {
160       return NextResponse.json(
161         { error: "Attendance not found for this class" },
162         { status: 400 }
163       );
164     }
165 
166     createdPresence = await prisma.attendancePresence.findFirst({
167       where: {
168         attendanceId,
169         studentId: createdStudent.id,
170       },
171       include: {
172         student: {
173           select: {
174             id: true,
175             name: true,
176           },
177         },
178       },
179     });
180   }
181 
182   return NextResponse.json(
183     {
184       student: createdStudent,
185       presence: createdPresence,
186     },
187     { status: 201 }
188   );
189 }
