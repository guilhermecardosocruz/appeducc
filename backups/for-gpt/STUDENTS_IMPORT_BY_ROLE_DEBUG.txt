1 import { NextRequest, NextResponse } from "next/server";
2 import { getSessionUser } from "@/lib/auth";
3 import { prisma } from "@/lib/prisma";
4 
5 async function ensureClassAccess(userId: string, classId: string) {
6   const foundClass = await prisma.class.findUnique({
7     where: { id: classId },
8     include: {
9       school: true,
10     },
11   });
12 
13   if (!foundClass) return null;
14 
15   const membership = await prisma.schoolMember.findUnique({
16     where: {
17       userId_schoolId: {
18         userId,
19         schoolId: foundClass.schoolId,
20       },
21     },
22   });
23 
24   if (!membership) return null;
25 
26   return foundClass;
27 }
28 
29 export async function POST(
30   req: NextRequest,
31   { params }: { params: Promise<{ classId: string }> }
32 ) {
33   const user = await getSessionUser();
34 
35   if (!user) {
36     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
37   }
38 
39   const { classId } = await params;
40   const foundClass = await ensureClassAccess(user.id, classId);
41 
42   if (!foundClass) {
43     return NextResponse.json({ error: "Forbidden" }, { status: 403 });
44   }
45 
46   const data = await req.json();
47   const name = String(data.name ?? "").trim();
48   const attendanceId = String(data.attendanceId ?? "").trim();
49 
50   if (!name) {
51     return NextResponse.json({ error: "Missing student name" }, { status: 400 });
52   }
53 
54   const createdStudent = await prisma.student.create({
55     data: {
56       name,
57       classId,
58     },
59   });
60 
61   let createdPresence = null;
62 
63   if (attendanceId) {
64     const attendance = await prisma.attendance.findUnique({
65       where: { id: attendanceId },
66     });
67 
68     if (!attendance || attendance.classId !== classId) {
69       return NextResponse.json(
70         { error: "Attendance not found for this class" },
71         { status: 400 }
72       );
73     }
74 
75     createdPresence = await prisma.attendancePresence.create({
76       data: {
77         attendanceId,
78         studentId: createdStudent.id,
79         present: false,
80       },
81       include: {
82         student: {
83           select: {
84             id: true,
85             name: true,
86           },
87         },
88       },
89     });
90   }
91 
92   return NextResponse.json(
93     {
94       student: createdStudent,
95       presence: createdPresence,
96     },
97     { status: 201 }
98   );
99 }
