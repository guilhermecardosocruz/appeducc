1 import { notFound } from "next/navigation";
2 import { getSessionUser } from "@/lib/auth";
3 import { prisma } from "@/lib/prisma";
4 import ClassDetailClient from "@/components/ClassDetailClient";
5 
6 type PageProps = {
7   params: Promise<{ classId: string }>;
8 };
9 
10 export default async function ClassPage({ params }: PageProps) {
11   const user = await getSessionUser();
12 
13   if (!user) {
14     notFound();
15   }
16 
17   const { classId } = await params;
18 
19   const foundClass = await prisma.class.findUnique({
20     where: { id: classId },
21     include: {
22       school: true,
23     },
24   });
25 
26   if (!foundClass) {
27     notFound();
28   }
29 
30   const membership = await prisma.schoolMember.findUnique({
31     where: {
32       userId_schoolId: {
33         userId: user.id,
34         schoolId: foundClass.schoolId,
35       },
36     },
37   });
38 
39   if (!membership) {
40     notFound();
41   }
42 
43   return (
44     <ClassDetailClient
45       classId={foundClass.id}
46       className={foundClass.name}
47       schoolId={foundClass.schoolId}
48     />
49   );
50 }
