5 function isManagerRole(role: string | null | undefined) {
6   const normalized = String(role ?? "").trim().toUpperCase();
7   return ["ADMIN", "GESTOR", "MANAGER", "OWNER"].includes(normalized);
8 }

10 async function getAccessContext(userId: string, classId: string) {
11   const foundClass = await prisma.class.findUnique({
12     where: { id: classId },
13     include: {
14       school: true,
15     },
16   });
17 
18   if (!foundClass) return null;
19 
20   const [schoolMembership, groupMembership] = await Promise.all([
21     prisma.schoolMember.findUnique({
22       where: {
23         userId_schoolId: {
24           userId,
25           schoolId: foundClass.schoolId,
26         },
27       },
28     }),
29     prisma.groupMember.findUnique({
30       where: {
31         userId_groupId: {
32           userId,
33           groupId: foundClass.school.groupId,
34         },
35       },
36     }),
37   ]);
38 
39   const hasAccess = Boolean(schoolMembership);
40   const canManageDelete =
41     isManagerRole(schoolMembership?.role) || isManagerRole(groupMembership?.role);
42 
43   return {
44     foundClass,
45     hasAccess,
46     canManageDelete,
47   };
48 }

50 export async function GET(
51   _req: NextRequest,
52   { params }: { params: Promise<{ classId: string; studentId: string }> }
53 ) {
54   const user = await getSessionUser();
55   if (!user) {
56     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
57   }
58 
59   const { classId, studentId } = await params;

61

64   }
65

67     where: { id: studentId },
68     include: {
69       presences: true,
70     },
71   });
72

74     return NextResponse.json({ error: "Student not found" }, { status: 404 });
75   }
76 
77   const total = student.presences.length;
78   const presents = student.presences.filter((p) => p.present).length;
79   const absents = total - presents;
80   const percentage = total ? Math.round((presents / total) * 100) : 0;
81 
82   return NextResponse.json({
83     id: student.id,
84     name: student.name,
85     status: student.status,
86     deletedAt: student.deletedAt,
87     deletedReason: student.deletedReason,
88     canApproveDelete: access.canManageDelete,
89     stats: {

91       presents,
92       absents,
93       percentage,
94     },
95   });

141 export async function DELETE(
142   req: NextRequest,
143   { params }: { params: Promise<{ classId: string; studentId: string }> }
144 ) {
145   const user = await getSessionUser();
146   if (!user) {
147     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
148   }
149

151   const access = await getAccessContext(user.id, classId);
152 
153   if (!access?.hasAccess) {
154     return NextResponse.json({ error: "Forbidden" }, { status: 403 });
155   }
156

158   const reason = String(body?.reason ?? "").trim();

163       { status: 400 }
164     );
165   }

168     where: { id: studentId },

170       id: true,
171       classId: true,

173     },
174   });
175 
176   if (!student || student.classId !== classId) {
177     return NextResponse.json({ error: "Student not found" }, { status: 404 });
178   }
179 
180   if (access.canManageDelete) {
181     await prisma.student.update({
182       where: { id: studentId },
183       data: {
184         status: "DELETED",
185         deletedReason: reason,
186         deletedById: user.id,
187         deletedAt: new Date(),
188       },
189     });
190 
191     return NextResponse.json({
192       success: true,
193       mode: "deleted_directly",
194     });
195   }
196 
197   await prisma.student.update({
198     where: { id: studentId },
199     data: {
200       status: "PENDING_DELETE",
201       deletedReason: reason,
202       deletedById: user.id,
203       deletedAt: null,
204     },
205   });
206 
207   return NextResponse.json({
208     success: true,
209     mode: "requested_delete",
210   });
211 }
