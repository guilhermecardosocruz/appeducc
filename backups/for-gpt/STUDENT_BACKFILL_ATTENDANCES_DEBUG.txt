30 function extractNamesFromWorkbook(buffer: Buffer, filename: string) {
31   const workbook = XLSX.read(buffer, { type: "buffer" });
32   const firstSheetName = workbook.SheetNames[0];
33 
34   if (!firstSheetName) {
35     return [];
36   }
37 
38   const sheet = workbook.Sheets[firstSheetName];
39   const rows = XLSX.utils.sheet_to_json<(string | number | null)[]>(sheet, {
40     header: 1,
41     raw: false,
42     blankrows: false,
43   });
44 
45   if (!rows.length) {
46     return [];
47   }
48 
49   const normalizedFilename = filename.toLowerCase();
50   const firstRow = rows[0] ?? [];
51   const headerIndex = firstRow.findIndex((cell) =>
52     String(cell ?? "")
53       .trim()
54       .toLowerCase()
55       .match(/^(nome|aluno|student|name)$/)
56   );
57 
58   const hasHeader = headerIndex >= 0;
59 
60   if (hasHeader) {
61     return rows
62       .slice(1)
63       .map((row) => String(row[headerIndex] ?? "").trim())
64       .filter(Boolean);
65   }
66 
67   return rows
68     .map((row) => String((row[0] ?? "") as string).trim())
69     .filter(Boolean)
70     .filter((value) => {
71       if (!normalizedFilename.endsWith(".csv")) return true;
72       return value.toLowerCase() !== "nome";
73     });
74 }

76 export async function POST(
77   req: NextRequest,
78   { params }: { params: Promise<{ classId: string }> }
79 ) {
80   const user = await getSessionUser();
81 
82   if (!user) {
83     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
84   }
85 
86   if (user.isTeacher) {
87     return NextResponse.json(
88       { error: "Only managers can import spreadsheets" },
89       { status: 403 }
90     );
91   }
92 
93   const { classId } = await params;
94   const foundClass = await ensureClassAccess(user.id, classId);
95 
96   if (!foundClass) {
97     return NextResponse.json({ error: "Forbidden" }, { status: 403 });
98   }
99 
100   const formData = await req.formData();
101   const file = formData.get("file");
102 
103   if (!(file instanceof File)) {
104     return NextResponse.json({ error: "Missing file" }, { status: 400 });
105   }
106 
107   const filename = file.name || "students.xlsx";
108   const bytes = await file.arrayBuffer();
109   const buffer = Buffer.from(bytes);
110 
111   const extractedNames = extractNamesFromWorkbook(buffer, filename)
112     .map((name) => name.trim())
113     .filter(Boolean);
114 
115   if (extractedNames.length === 0) {
116     return NextResponse.json(
117       { error: "No student names found in file" },
118       { status: 400 }
119     );
120   }
121 
122   const existingStudents = await prisma.student.findMany({
123     where: { classId },
124     select: { name: true },
125   });
126 
127   const existingNameSet = new Set(
128     existingStudents.map((student) => student.name.trim().toLowerCase())
129   );
130 
131   const uniqueNamesToCreate: string[] = [];
132   const seenInFile = new Set<string>();
133   let skipped = 0;
134 
135   for (const rawName of extractedNames) {
136     const normalized = rawName.trim().toLowerCase();
137 
138     if (!normalized) {
139       skipped += 1;
140       continue;
141     }
142 
143     if (existingNameSet.has(normalized) || seenInFile.has(normalized)) {
144       skipped += 1;
145       continue;
146     }
147 
148     seenInFile.add(normalized);
149     uniqueNamesToCreate.push(rawName.trim());
150   }
151 
152   if (uniqueNamesToCreate.length > 0) {
153     await prisma.student.createMany({
154       data: uniqueNamesToCreate.map((name) => ({
155         name,
156         classId,
157       })),
158     });
159   }
160 
161   const students = await prisma.student.findMany({
162     where: { classId },
163     orderBy: { name: "asc" },
164   });
165 
166   return NextResponse.json({
167     imported: uniqueNamesToCreate.length,
168     skipped,
169     students,
170   });
171 }

244         onClose={() => setOpenStudentModal(false)}
245         onSubmit={handleAddStudent}
246         loading={addingStudent}
