29 export async function DELETE(
30   req: NextRequest,
31   { params }: { params: Promise<{ classId: string; studentId: string }> }
32 ) {
33   const user = await getSessionUser();
34 
35   if (!user) {
36     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
37   }
38 
39   const { classId, studentId } = await params;
40 
41   const foundClass = await ensureClassAccess(user.id, classId);
42 
43   if (!foundClass) {
44     return NextResponse.json({ error: "Forbidden" }, { status: 403 });
45   }
46 
47   const data = await req.json();
48   const reason = String(data.reason ?? "").trim();
49 
50   if (!reason) {
51     return NextResponse.json(
52       { error: "Deletion reason is required" },
53       { status: 400 }
54     );
55   }
56 
57   const student = await prisma.student.findUnique({
58     where: { id: studentId },
59   });
60 
61   if (!student || student.classId !== classId) {
62     return NextResponse.json({ error: "Student not found" }, { status: 404 });
63   }
64 
65   await prisma.student.update({
66     where: { id: studentId },
67     data: {
68       deletedAt: new Date(),
69       deletedReason: reason,
70       deletedById: user.id,
71     },
72   });
73 
74   return NextResponse.json({ success: true });
75 }

213                       onChange={() => togglePresence(item.id)}
214                       className="h-5 w-5 rounded border-slate-300 text-sky-600 focus:ring-sky-500"
215                     />
216                   </div>
217                 </li>
218               ))}
219             </ul>
220           )}
221 
222           <div className="flex justify-between gap-3">
223             <Link
224               href={`/classes/${classId}/chamadas`}
225               className="inline-flex items-center rounded-md border border-slate-300 bg-white px-4 py-2 text-sm font-medium text-slate-800 hover:bg-slate-50"
226             >
227               Voltar
228             </Link>
229 
230             <button
231               type="button"
232               onClick={handleSave}
233               disabled={loading}
234               className="inline-flex items-center rounded-md bg-sky-600 px-4 py-2 text-sm font-medium text-white shadow-sm hover:bg-sky-700 disabled:opacity-50"
235             >
236               {loading ? "Salvando..." : "Salvar chamada"}
237             </button>
238           </div>
239         </div>
240       </div>
241 
242       <AddStudentModal
243         open={openStudentModal}
244         onClose={() => setOpenStudentModal(false)}
245         onSubmit={handleAddStudent}
246         loading={addingStudent}
247       />
248     </>
249   );
