9 function generateTeacherPassword(name: string, cpf: string) {

62 export async function POST(req: NextRequest) {
63   const user = await getSessionUser();
64 
65   if (!user) {
66     return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
67   }
68 
69   const data = await req.json();
70   const name = String(data.name ?? "").trim();
71   const email = String(data.email ?? "").toLowerCase().trim();
72   const cpfRaw = String(data.cpf ?? "").trim();
73   const cpf = normalizeCpf(cpfRaw);
74 
75   if (!name || !email || !cpf) {
76     return NextResponse.json(
77       { error: "Missing teacher name, email or CPF" },
78       { status: 400 }
79     );
80   }
81 
82   if (cpf.length !== 11) {
83     return NextResponse.json(
84       { error: "CPF must contain 11 digits" },
85       { status: 400 }
86     );
87   }
88 
89   const existingByEmail = await prisma.user.findUnique({
90     where: { email },
91   });
92 
93   // Se usuário já existe → não altera senha
94   if (existingByEmail) {
95     const updated = await prisma.user.update({
96       where: { id: existingByEmail.id },
97       data: {
98         isTeacher: true,
99         createdById: user.id,
100         cpf: existingByEmail.cpf ?? cpf,
101       },
102       include: {
103         _count: {
104           select: {
105             classes: true,
